def convert_escape_characters(text)
  result = en_et_wb_cec(text)
  result.gsub!(/\eT\[(.+)\]/i) do
    if $game_text.keys.include?($1)
      convert_escape_characters($game_text[$1][1])
    else
      "Invalid key [#{$1}]. No matching text exists."
    end
  end
  result
end
